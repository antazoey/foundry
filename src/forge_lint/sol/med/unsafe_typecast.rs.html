11declare_forge_lint!(
12    UNSAFE_TYPECAST,
13    Severity::Med,
14    "unsafe-typecast",
15    "typecasts that can truncate values should be checked"
16);
17
18impl<'hir> LateLintPass<'hir> for UnsafeTypecast {
19    fn check_expr(
20        &mut self,
21        ctx: &LintContext,
22        hir: &'hir hir::Hir<'hir>,
23        expr: &'hir hir::Expr<'hir>,
24    ) {
25        // Check for type cast expressions: Type(value)
26        if let ExprKind::Call(call, args, _) = &expr.kind
27            && let ExprKind::Type(hir::Type { kind: TypeKind::Elementary(ty), .. }) = &call.kind
28            && args.len() == 1
29            && let Some(call_arg) = args.exprs().next()
30            && is_unsafe_typecast_hir(hir, call_arg, ty)
31        {
32            ctx.emit_with_fix(
33                &UNSAFE_TYPECAST,
34                expr.span,
35                Snippet::Block {
36                    desc: Some("Consider disabling this lint if you're certain the cast is safe:"),
37                    code: format!(
38                        "// casting to '{abi_ty}' is safe because [explain why]\n// forge-lint: disable-next-line(unsafe-typecast)",
39                        abi_ty = ty.to_abi_str()
40                    )
41                }
42            );
43        }
44    }
45}

47/// Determines if a typecast is potentially unsafe (could lose data or precision).
48fn is_unsafe_typecast_hir(
49    hir: &hir::Hir<'_>,
50    source_expr: &hir::Expr<'_>,
51    target_type: &hir::ElementaryType,
52) -> bool {
53    let mut source_types = Vec::<ElementaryType>::new();
54    infer_source_types(Some(&mut source_types), hir, source_expr);
55
56    if source_types.is_empty() {
57        return false;
58    };
59
60    source_types.iter().any(|source_ty| is_unsafe_elementary_typecast(source_ty, target_type))
61}

63/// Infers the elementary source type(s) of an expression.
64///
65/// This function traverses an expression tree to find the original "source" types.
66/// For cast chains, it returns the ultimate source type, not intermediate cast results.
67/// For binary operations, it collects types from both sides into the `output` vector.
68///
69/// # Returns
70/// An `Option<ElementaryType>` containing the inferred type of the expression if it can be
71/// resolved to a single source (like variables, literals, or unary expressions).
72/// Returns `None` for expressions complex expressions (like binary operations).
73fn infer_source_types(
74    mut output: Option<&mut Vec<ElementaryType>>,
75    hir: &hir::Hir<'_>,
76    expr: &hir::Expr<'_>,
77) -> Option<ElementaryType> {
78    let mut track = |ty: ElementaryType| -> Option<ElementaryType> {
79        if let Some(output) = output.as_mut() {
80            output.push(ty);
81        }
82        Some(ty)
83    };
84
85    match &expr.kind {
86        // A type cast call: `Type(val)`
87        ExprKind::Call(call_expr, args, ..) => {
88            // Check if the called expression is a type, which indicates a cast.
89            if let ExprKind::Type(hir::Type { kind: TypeKind::Elementary(..), .. }) =
90                &call_expr.kind
91                && let Some(inner) = args.exprs().next()
92            {
93                // Recurse to find the original (inner-most) source type.
94                return infer_source_types(output, hir, inner);
95            }
96            None
97        }
98
99        // Identifiers (variables)
100        ExprKind::Ident(resolutions) => {
101            if let Some(Res::Item(ItemId::Variable(var_id))) = resolutions.first() {
102                let variable = hir.variable(*var_id);
103                if let TypeKind::Elementary(elem_type) = &variable.ty.kind {
104                    return track(*elem_type);
105                }
106            }
107            None
108        }
109
110        // Handle literal values
111        ExprKind::Lit(hir::Lit { kind, .. }) => match kind {
112            LitKind::Str(StrKind::Hex, ..) => track(ElementaryType::Bytes),
113            LitKind::Str(..) => track(ElementaryType::String),
114            LitKind::Address(_) => track(ElementaryType::Address(false)),
115            LitKind::Bool(_) => track(ElementaryType::Bool),
116            // Unnecessary to check numbers as assigning literal values that cannot fit into a type
117            // throws a compiler error. Reference: <https://solang.readthedocs.io/en/latest/language/types.html>
118            _ => None,
119        },
120
121        // Unary operations: Recurse to find the source type of the inner expression.
122        ExprKind::Unary(_, inner_expr) => infer_source_types(output, hir, inner_expr),
123
124        // Binary operations
125        ExprKind::Binary(lhs, _, rhs) => {
126            if let Some(mut output) = output {
127                // Recurse on both sides to find and collect all source types.
128                infer_source_types(Some(&mut output), hir, lhs);
129                infer_source_types(Some(&mut output), hir, rhs);
130            }
131            None
132        }
133
134        // Complex expressions are not evaluated
135        _ => None,
136    }
137}

139/// Checks if a type cast from source_type to target_type is unsafe.
140fn is_unsafe_elementary_typecast(
141    source_type: &ElementaryType,
142    target_type: &ElementaryType,
143) -> bool {
144    match (source_type, target_type) {
145        // Numeric downcasts (smaller target size)
146        (ElementaryType::UInt(source_size), ElementaryType::UInt(target_size))
147        | (ElementaryType::Int(source_size), ElementaryType::Int(target_size)) => {
148            source_size.bits() > target_size.bits()
149        }
150
151        // Signed to unsigned conversion (potential loss of sign)
152        (ElementaryType::Int(_), ElementaryType::UInt(_)) => true,
153
154        // Unsigned to signed conversion with same or smaller size
155        (ElementaryType::UInt(source_size), ElementaryType::Int(target_size)) => {
156            source_size.bits() >= target_size.bits()
157        }
158
159        // Fixed bytes to smaller fixed bytes
160        (ElementaryType::FixedBytes(source_size), ElementaryType::FixedBytes(target_size)) => {
161            source_size.bytes() > target_size.bytes()
162        }
163
164        // Dynamic bytes to fixed bytes (potential truncation)
165        (ElementaryType::Bytes, ElementaryType::FixedBytes(_))
166        | (ElementaryType::String, ElementaryType::FixedBytes(_)) => true,
167
168        // Address to smaller uint (truncation) - address is 160 bits
169        (ElementaryType::Address(_), ElementaryType::UInt(target_size)) => target_size.bits() < 160,
170
171        // Address to int (sign issues)
172        (ElementaryType::Address(_), ElementaryType::Int(_)) => true,
173
174        _ => false,
175    }
